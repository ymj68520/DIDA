### 1.3 模块二先行：Solidity 智能合约开发

**注意**：合约必须在 BIND9 配置之前完成，因为 `TxID` 由合约部署逻辑决定，Zone File 中的 TxID 依赖合约状态。

**文件路径**：`contracts/src/IPCertRegistry.sol`

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title IPCertRegistry
 * @notice 存储 IP 凭证（Cert_IP）与顶级权威授权签名（Sig_Top）
 *         TxID -> ChainRecord 的扁平映射，单次 eth_call 返回完整凭证
 */
contract IPCertRegistry {

    // ── 凭证主体（Cert_IP），V1 验签的被签名对象 ──
    struct CertIP {
        string  ipPrefix;       // 被授权的 IP 前缀，如 "192.168.1.0/24"
        bytes   publicKey;      // 凭证持有者公钥 PK_Sub（secp256k1，65字节非压缩）
        uint64  expiration;     // 凭证过期 Unix 时间戳
        bool    isRevoked;      // 吊销状态位
    }

    // ── 链上完整记录 ──
    struct ChainRecord {
        CertIP  certIP;   // 凭证主体（被签名对象）
        bytes   sigTop;   // 顶级权威对 keccak256(abi.encode(certIP)) 的授权签名 Sig_Top
    }

    // TxID (bytes32) => 完整链上记录
    mapping(bytes32 => ChainRecord) public registry;

    // 权威管理员地址（实验阶段为合约部署者）
    address public immutable authority;

    // ── 事件（供网关 WebSocket 订阅）──
    event CertRegistered(bytes32 indexed txId, string ipPrefix);
    event Revoked(bytes32 indexed txId);

    constructor() {
        authority = msg.sender;
    }

    modifier onlyAuthority() {
        require(msg.sender == authority, "Not authorized");
        _;
    }

    /**
     * @notice 注册 IP 凭证（管理面，需顶级权威账户调用）
     * @param txId   32字节哈希指针（由 keccak256(ipPrefix + nonce) 生成，确保唯一性）
     * @param certIP 凭证主体
     * @param sigTop 顶级权威对 keccak256(abi.encode(certIP)) 的签名
     */
    function registerCert(
        bytes32        txId,
        CertIP calldata certIP,
        bytes  calldata sigTop
    ) external onlyAuthority {
        require(bytes(certIP.ipPrefix).length > 0, "Empty ipPrefix");
        require(certIP.publicKey.length == 65,      "Invalid publicKey length");
        require(certIP.expiration > block.timestamp, "Already expired");
        require(sigTop.length > 0,                   "Missing sigTop");

        registry[txId] = ChainRecord({ certIP: certIP, sigTop: sigTop });
        emit CertRegistered(txId, certIP.ipPrefix);
    }

    /**
     * @notice 吊销凭证（管理面）
     */
    function revokeCert(bytes32 txId) external onlyAuthority {
        require(bytes(registry[txId].certIP.ipPrefix).length > 0, "Record not found");
        registry[txId].certIP.isRevoked = true;
        emit Revoked(txId);
    }

    /**
     * @notice 数据面只读查询（网关调用 eth_call）
     */
    function getRecord(bytes32 txId) external view returns (ChainRecord memory) {
        return registry[txId];
    }
}
```

### 1.4 凭证生成脚本（离线签发 Sig_Top，写入链上）

**文件**：`scripts/provision_certs.py`

```python
#!/usr/bin/env python3
"""
离线生成 Cert_IP，用 SK_Top 签名生成 Sig_Top，然后调用合约 registerCert。
运行前需设置环境变量：
    RPC_URL         = http://127.0.0.1:8545
    CONTRACT_ADDR   = 0x...
    SK_TOP          = 0x...（顶级权威私钥，仅实验环境使用）
    SK_SUB_LIST     = 逗号分隔的多个私钥（各测试 IP 的持有者私钥）
"""
import os, json, hashlib, struct, time
from eth_account import Account
from eth_abi import encode
from web3 import Web3

Account.enable_unaudited_hdwallet_features()

RPC_URL       = os.environ["RPC_URL"]
CONTRACT_ADDR = os.environ["CONTRACT_ADDR"]
SK_TOP        = os.environ["SK_TOP"]
w3            = Web3(Web3.HTTPProvider(RPC_URL))

# 测试用 IP 前缀和对应的子密钥
TEST_CERTS = [
    {"ipPrefix": "192.168.1.0/24",  "sk_sub_env": "SK_SUB_1"},
    {"ipPrefix": "192.168.2.0/24",  "sk_sub_env": "SK_SUB_2"},
    {"ipPrefix": "10.0.0.0/8",      "sk_sub_env": "SK_SUB_3"},
    {"ipPrefix": "172.16.0.0/12",   "sk_sub_env": "SK_SUB_4"},
    {"ipPrefix": "203.0.113.0/24",  "sk_sub_env": "SK_SUB_5"},
]

# 加载合约 ABI
with open("contracts/out/IPCertRegistry.sol/IPCertRegistry.json") as f:
    abi = json.load(f)["abi"]
contract = w3.eth.contract(address=CONTRACT_ADDR, abi=abi)

authority = Account.from_key(SK_TOP)
results   = []   # 输出到 config/cert_manifest.json

for idx, entry in enumerate(TEST_CERTS):
    sk_sub = Account.from_key(os.environ[entry["sk_sub_env"]])
    pk_sub = bytes.fromhex(sk_sub.key.public_key.to_hex()[2:])  # 65字节
    expiration = int(time.time()) + 365 * 24 * 3600  # 一年后到期

    # 构造 CertIP（与 Solidity 结构体 ABI 编码一致）
    cert_ip_encoded = encode(
        ["(string,bytes,uint64,bool)"],
        [(entry["ipPrefix"], pk_sub, expiration, False)]
    )
    cert_ip_hash = Web3.keccak(cert_ip_encoded)

    # 用 SK_Top 签名 Hash(Cert_IP) → Sig_Top
    sig_top_obj = authority.sign_message(
        messages.encode_defunct(hexstr=cert_ip_hash.hex())
    )
    sig_top = sig_top_obj.signature  # bytes，65字节

    # 生成 TxID = keccak256(ipPrefix + expiration)
    tx_id_bytes = Web3.keccak(
        text=f"{entry['ipPrefix']}:{expiration}:{idx}"
    )

    # 生成 Sig_Sub = Sign(SK_Sub, IP_prefix || TxID)
    msg_for_sig_sub = (entry["ipPrefix"] + tx_id_bytes.hex()).encode()
    sig_sub_obj = sk_sub.sign_message(messages.encode_defunct(msg_for_sig_sub))
    sig_sub = sig_sub_obj.signature.hex()

    # 调用合约 registerCert
    cert_ip_struct = (entry["ipPrefix"], pk_sub, expiration, False)
    tx = contract.functions.registerCert(
        tx_id_bytes, cert_ip_struct, sig_top
    ).build_transaction({
        "from": authority.address,
        "nonce": w3.eth.get_transaction_count(authority.address),
        "gas": 300000,
    })
    signed = authority.sign_transaction(tx)
    tx_hash = w3.eth.send_raw_transaction(signed.rawTransaction)
    w3.eth.wait_for_transaction_receipt(tx_hash)

    results.append({
        "txId":      tx_id_bytes.hex(),
        "ipPrefix":  entry["ipPrefix"],
        "sig_sub":   sig_sub,    # 写入 DNS Zone File
        "pk_sub":    pk_sub.hex(),
    })
    print(f"[OK] Registered {entry['ipPrefix']} → TxID: {tx_id_bytes.hex()[:16]}...")

with open("config/cert_manifest.json", "w") as f:
    json.dump(results, f, indent=2)
print("\n[Done] cert_manifest.json written.")
```

**1.5.3 Zone 文件生成脚本（读取 cert_manifest.json 自动生成）**

文件：`scripts/gen_zone.py`

```python
#!/usr/bin/env python3
"""
根据 config/cert_manifest.json 自动生成 BIND9 Zone File。
输出：/etc/bind/zones/db.reverse.test
"""
import json, ipaddress, os

with open("config/cert_manifest.json") as f:
    certs = json.load(f)

zone_lines = [
    "$TTL 3600",
    "@ IN SOA ns1.test.local. admin.test.local. (",
    "         2024010101 ; serial",
    "         3600       ; refresh",
    "         900        ; retry",
    "         604800     ; expire",
    "         300 )      ; minimum TTL",
    "  IN NS  ns1.test.local.",
    "",
]

for cert in certs:
    prefix = cert["ipPrefix"]
    tx_id  = cert["txId"]
    sig_sub = cert["sig_sub"]

    # 对 /24 前缀：提取网络地址的最后一段作为反向记录前缀
    net    = ipaddress.ip_network(prefix, strict=False)
    # 取网络地址的末尾主机号范围，这里为演示写一个代表性主机记录
    # 生产中应枚举该前缀下所有需要认证的 IP
    host   = str(net.network_address).split(".")[-1]   # 取第4段

    # TXT 记录格式：v=rOBD; txid=0x...; sig_sub=0x...
    txt_record = f'"v=rOBD; txid={tx_id}; sig_sub={sig_sub}"'

    # PTR + TXT 记录
    zone_lines.append(f"; {prefix}")
    zone_lines.append(f"{host} IN PTR server-{prefix.replace('/', '-').replace('.', '-')}.test.local.")
    zone_lines.append(f"{host} IN TXT {txt_record}")
    zone_lines.append("")

os.makedirs("/etc/bind/zones", exist_ok=True)
zone_content = "\n".join(zone_lines)
with open("/etc/bind/zones/db.reverse.test", "w") as f:
    f.write(zone_content)

print("[OK] Zone file written to /etc/bind/zones/db.reverse.test")
print("     Restart BIND9 to apply: sudo systemctl restart bind9")
```

### 2.1 批量灌注脚本（支撑 Exp-3：存储规模与时延关系）

```bash
#!/usr/bin/env bash
# 分批向合约灌注 100 / 1,000 / 10,000 条凭证
# 每批次完成后记录合约状态根哈希（用于 Exp-3 的数据点）

set -e
source config/contract.env  # 导入 CONTRACT_ADDR, RPC_URL, SK_TOP
```

### 3.4 流水线核心：`src/pipeline/mod.rs`

```rust
use crate::state::{GatewayState, ChainRecord};
use crate::telemetry::PipelineTimer;
use std::{net::IpAddr, sync::Arc, time::{SystemTime, UNIX_EPOCH}};
use eyre::Result;

/// 处理单个 TCP SYN 报文的完整流水线
/// 返回 true → NF_ACCEPT；返回 false → NF_DROP
pub async fn process_packet(
    state: Arc<GatewayState>,
    dst_ip: IpAddr,
    src_ip: IpAddr,
) -> bool {
    let timer = PipelineTimer::start();

    // ── In-Flight 去重（防 TCP 重传） ──────────────────────────
    let key = (src_ip, dst_ip);
    if state.inflight.get(&key).await.is_some() {
        tracing::debug!("In-flight duplicate, skipping: {:?}", key);
        return false;
    }
    state.inflight.insert(key, ()).await;

    let result = process_inner(&state, dst_ip, &timer).await;

    state.inflight.invalidate(&key).await;
    timer.record_total();
    result
}

async fn process_inner(
    state: &Arc<GatewayState>,
    dst_ip: IpAddr,
    timer: &PipelineTimer,
) -> bool {
    // ── Step 1: 带外 DNS 查询（获取 TxID + Sig_Sub） ────────────
    let dns_result = tokio::time::timeout(
        std::time::Duration::from_millis(50),
        crate::pipeline::dns::query_txt(state, dst_ip),
    ).await;

    let (tx_id, sig_sub) = match dns_result {
        Ok(Ok(v))  => v,
        Ok(Err(e)) => { tracing::warn!("DNS failed: {e}"); return false; }
        Err(_)     => { tracing::warn!("DNS timeout");      return false; }
    };
    timer.record_dns();

    // ── Step 2: 链上状态查询（Cache 优先，Miss 则 RPC） ─────────
    let chain_record: Arc<ChainRecord> = match state.cert_cache.get(&tx_id).await {
        Some(record) => {
            timer.record_cache_hit();
            record
        }
        None => {
            let rpc_result = tokio::time::timeout(
                std::time::Duration::from_millis(200),
                crate::pipeline::chain::fetch_record(state, tx_id),
            ).await;

            match rpc_result {
                Ok(Ok(r))  => {
                    let r = Arc::new(r);
                    state.cert_cache.insert(tx_id, r.clone()).await;
                    timer.record_rpc();
                    r
                }
                Ok(Err(e)) => { tracing::warn!("RPC failed: {e}");  return false; }
                Err(_)     => { tracing::warn!("RPC timeout");       return false; }
            }
        }
    };

    // ── Step 3: V1 校验（用本地 PK_Top 验 Sig_Top） ─────────────
    // V1 使用本地预置 PK_Top，无需网络，在 spawn_blocking 中执行密码学运算
    let record_clone  = chain_record.clone();
    let pk_top_clone  = state.pk_top.clone();
    let dst_ip_clone  = dst_ip;

    let v1_ok = tokio::task::spawn_blocking(move || {
        crate::pipeline::crypto::verify_v1(
            &pk_top_clone,
            &record_clone,
            dst_ip_clone,
        )
    }).await.unwrap_or(false);

    timer.record_v1();

    if !v1_ok {
        tracing::warn!("V1 failed for {dst_ip}");
        return false;
    }

    // ── Step 4: V2 校验（从 Cert_IP 提取 PK_Sub 验 Sig_Sub） ────
    let pk_sub   = record_clone.cert_ip.public_key.clone();
    let v2_ok = tokio::task::spawn_blocking(move || {
        crate::pipeline::crypto::verify_v2(
            &pk_sub,
            &sig_sub,
            dst_ip_clone,
            &tx_id,
        )
    }).await.unwrap_or(false);

    timer.record_v2();

    if !v2_ok {
        tracing::warn!("V2 failed for {dst_ip}");
        return false;
    }

    tracing::info!("Auth OK for {dst_ip}");
    true
}
```

### 3.5 密码学验证：`src/pipeline/crypto.rs`

```rust
use crate::state::{ChainRecord, CertIP};
use k256::{
    ecdsa::{Signature, VerifyingKey, signature::Verifier},
    sha2::Sha256,
};
use sha3::{Keccak256, Digest};
use std::{net::IpAddr, time::{SystemTime, UNIX_EPOCH}};
use ipnet::IpNet;

/// V1 校验：
///   1. Verify(PK_Top, Sig_Top, keccak256(abi_encode(CertIP)))
///   2. T_current <= CertIP.expiration
///   3. dst_ip ∈ CertIP.ip_prefix
///   4. !CertIP.is_revoked
pub fn verify_v1(
    pk_top:  &VerifyingKey,
    record:  &ChainRecord,
    dst_ip:  IpAddr,
) -> bool {
    // 1. 计算 Hash(Cert_IP)
    let cert_hash = keccak256_cert_ip(&record.cert_ip);

    // 2. 验证 Sig_Top
    let Ok(sig) = Signature::from_slice(&record.sig_top) else {
        return false;
    };
    if pk_top.verify(&cert_hash, &sig).is_err() {
        return false;
    }

    // 3. 时间有效性
    let now = SystemTime::now()
        .duration_since(UNIX_EPOCH).unwrap().as_secs();
    if now > record.cert_ip.expiration_timestamp {
        return false;
    }

    // 4. IP 归属校验
    if !record.cert_ip.ip_prefix.contains(&dst_ip) {
        return false;
    }

    // 5. 吊销状态
    !record.cert_ip.is_revoked
}

/// V2 校验：Verify(PK_Sub, Sig_Sub, IP_bytes || TxID_bytes)
pub fn verify_v2(
    pk_sub:  &VerifyingKey,
    sig_sub: &[u8],
    dst_ip:  IpAddr,
    tx_id:   &[u8; 32],
) -> bool {
    // 构造被签名消息：IP 字节表示 || TxID
    let mut message = Vec::new();
    match dst_ip {
        IpAddr::V4(v4) => message.extend_from_slice(&v4.octets()),
        IpAddr::V6(v6) => message.extend_from_slice(&v6.octets()),
    }
    message.extend_from_slice(tx_id);

    let msg_hash = Keccak256::digest(&message);

    let Ok(sig) = Signature::from_slice(sig_sub) else {
        return false;
    };
    pk_sub.verify(msg_hash.as_slice(), &sig).is_ok()
}

/// 计算 Cert_IP 的 Keccak256 哈希（与 Solidity 的 keccak256(abi.encode(certIP)) 对齐）
fn keccak256_cert_ip(cert: &CertIP) -> Vec<u8> {
    let mut hasher = Keccak256::new();
    hasher.update(cert.ip_prefix.to_string().as_bytes());
    let pk_bytes = cert.public_key.to_encoded_point(false);
    hasher.update(pk_bytes.as_bytes());
    hasher.update(&cert.expiration_timestamp.to_be_bytes());
    hasher.update(&[cert.is_revoked as u8]);
    hasher.finalize().to_vec()
}
```

### 3.6 遥测探针：`src/telemetry.rs`

```rust
use std::time::Instant;
use std::sync::atomic::{AtomicU64, Ordering};

/// 每个请求独立的时延探针（栈上分配，零堆开销）
pub struct PipelineTimer {
    start:     Instant,
    dns_done:  Option<Instant>,
    rpc_done:  Option<Instant>,
    cache_hit: bool,
    v1_done:   Option<Instant>,
    v2_done:   Option<Instant>,
}

impl PipelineTimer {
    pub fn start() -> Self {
        PipelineTimer {
            start: Instant::now(),
            dns_done: None, rpc_done: None,
            cache_hit: false,
            v1_done: None,  v2_done: None,
        }
    }

    pub fn record_dns(&mut self)       { self.dns_done  = Some(Instant::now()); }
    pub fn record_rpc(&mut self)       { self.rpc_done  = Some(Instant::now()); }
    pub fn record_cache_hit(&mut self) { self.cache_hit = true; self.rpc_done = Some(Instant::now()); }
    pub fn record_v1(&mut self)        { self.v1_done   = Some(Instant::now()); }
    pub fn record_v2(&mut self)        { self.v2_done   = Some(Instant::now()); }

    /// 请求结束时输出 CSV 行：total_us,dns_us,rpc_us,v1_us,v2_us,cache_hit
    pub fn record_total(&self) {
        let now   = Instant::now();
        let total = now.duration_since(self.start).as_micros();
        let dns   = self.dns_done.map(|t| t.duration_since(self.start).as_micros()).unwrap_or(0);
        let rpc   = self.rpc_done.map(|t| t.duration_since(self.dns_done.unwrap_or(self.start)).as_micros()).unwrap_or(0);
        let v1    = self.v1_done.map(|t| t.duration_since(self.rpc_done.unwrap_or(self.start)).as_micros()).unwrap_or(0);
        let v2    = self.v2_done.map(|t| t.duration_since(self.v1_done.unwrap_or(self.start)).as_micros()).unwrap_or(0);

        // 非阻塞写入内存缓冲区（通过全局 channel 收集，测试结束后 flush）
        TELEMETRY_SENDER.with(|tx| {
            let _ = tx.borrow().send(TelemetryRow { total, dns, rpc, v1, v2, cache_hit: self.cache_hit });
        });
    }
}
```

### 4.2 完整握手压测客户端：`src/bin/load_client.rs`

```rust
//! Exp-5 压测客户端：完整 TCP 三次握手并发认证
//! 用法：cargo run --bin load_client -- --target 192.168.1.100:80 --concurrency 1000 --duration 30

use std::{net::SocketAddr, time::Duration};
use tokio::{net::TcpStream, time::Instant};
use clap::Parser;
use std::sync::{Arc, atomic::{AtomicU64, Ordering}};

#[derive(Parser, Debug)]
struct Args {
    #[arg(long)] target:      SocketAddr,
    #[arg(long, default_value = "100")]  concurrency: usize,
    #[arg(long, default_value = "30")]   duration:    u64,
    #[arg(long, default_value = "results/exp5/raw.csv")] output: String,
}

#[tokio::main]
async fn main() -> eyre::Result<()> {
    let args       = Args::parse();
    let success    = Arc::new(AtomicU64::new(0));
    let failure    = Arc::new(AtomicU64::new(0));
    let latencies  = Arc::new(tokio::sync::Mutex::new(Vec::<f64>::with_capacity(1_000_000)));

    let deadline = Instant::now() + Duration::from_secs(args.duration);

    // 创建指定并发度的 tokio 任务
    let mut handles = Vec::new();
    for _ in 0..args.concurrency {
        let target    = args.target;
        let success   = success.clone();
        let failure   = failure.clone();
        let lats      = latencies.clone();

        handles.push(tokio::spawn(async move {
            while Instant::now() < deadline {
                let t0  = Instant::now();
                // 完整 TCP 三次握手（connect = SYN → SYN-ACK → ACK）
                match TcpStream::connect(target).await {
                    Ok(stream) => {
                        let lat_ms = t0.elapsed().as_secs_f64() * 1000.0;
                        success.fetch_add(1, Ordering::Relaxed);
                        lats.lock().await.push(lat_ms);
                        drop(stream);  // 立即关闭连接（发 FIN）
                    }
                    Err(_) => { failure.fetch_add(1, Ordering::Relaxed); }
                }
            }
        }));
    }

    for h in handles { let _ = h.await; }

    // 汇总与输出
    let lats_snapshot = latencies.lock().await.clone();
    let total  = success.load(Ordering::Relaxed) + failure.load(Ordering::Relaxed);
    let tps    = success.load(Ordering::Relaxed) as f64 / args.duration as f64;

    let mut sorted = lats_snapshot.clone();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let p50 = sorted.get(sorted.len() / 2).copied().unwrap_or(0.0);
    let p95 = sorted.get(sorted.len() * 95 / 100).copied().unwrap_or(0.0);
    let p99 = sorted.get(sorted.len() * 99 / 100).copied().unwrap_or(0.0);

    println!("=== Exp-5 结果 ===");
    println!("并发度: {}  持续时间: {}s", args.concurrency, args.duration);
    println!("成功:  {}  失败:  {}  TPS: {:.1}", success.load(Ordering::Relaxed), failure.load(Ordering::Relaxed), tps);
    println!("P50: {:.2}ms  P95: {:.2}ms  P99: {:.2}ms", p50, p95, p99);

    // 写入 CSV（测试结束后一次性 flush）
    std::fs::create_dir_all(std::path::Path::new(&args.output).parent().unwrap())?;
    let mut w = csv::Writer::from_path(&args.output)?;
    w.write_record(&["latency_ms"])?;
    for lat in &lats_snapshot { w.write_record(&[format!("{:.4}", lat)])?; }
    w.flush()?;

    Ok(())
}
```
